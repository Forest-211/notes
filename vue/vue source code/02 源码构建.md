## 构建脚本

通常一个基于 NPM 托管的项目都会有一个 `package.json `文件，它是对项目的描述文件，它的内容实际上是一个标准的 JSON 对象。

## 构建过程

通过`package.json`中`scripts`对象的`build`命令（`node scripts/build.js`）分析，它执行的是项目根目录下scripts目录中的`build.js`的逻辑；核心代码如下：

```js
// 获取所有的配置
let builds = require("./config").getAllBuilds();

// 通过命令行参数过滤不需要打包的配置
if (process.argv[2]) {
  const filters = process.argv[2].split(",");
  builds = builds.filter((b) => {
    return filters.some(
      (f) => b.output.file.indexOf(f) > -1 || b._name.indexOf(f) > -1
    );
  });
} else {
  // 默认过滤掉 weex 构建
  builds = builds.filter((b) => {
    return b.output.file.indexOf("weex") === -1;
  });
}

// 执行构建逻辑
build(builds);

// 构建的核心逻辑
function build(builds) {
  // ...
}
```

上面的执行流程一目了然，先是从配置文件中读取配置，然后通过命令行参数过滤不希要的打包配置，然后就执行构建逻辑；接着来看看配置文件（scripts/config.js）有哪些配置；核心部分代码如下：

function genConfig(name) {
  const opts = builds[name];
  const config = {
    input: opts.entry,
    external: opts.external,
    plugins: [
      replace({
        __WEEX__: !!opts.weex,
        __WEEX_VERSION__: weexVersion,
        __VERSION__: version,
      }),
      flow(),
      buble(),
      alias(Object.assign({}, aliases, opts.alias)),
    ].concat(opts.plugins || []),
    output: {
      file: opts.dest,
      format: opts.format,
      banner: opts.banner,
      name: opts.moduleName || "Vue",
    },
  };

  if (opts.env) {
    config.plugins.push(
      replace({
        "process.env.NODE_ENV": JSON.stringify(opts.env),
      })
    );
  }

  Object.defineProperty(config, "_name", {
    enumerable: false,
    value: name,
  });

  return config;
}

if (process.env.TARGET) {
  module.exports = genConfig(process.env.TARGET);
} else {
  exports.getBuild = genConfig;
  exports.getAllBuilds = () => Object.keys(builds).map(genConfig);

对于单个配置，它是遵循 Rollup 的构建规则的。其中

- `entry` 属性表示构建的入口 JS 文件地址，
- `dest` 属性表示构建后的 JS 文件地址。
- `format` 属性表示构建的格式，
- `cjs` 表示构建出来的文件遵循 [CommonJS](http://wiki.commonjs.org/wiki/Modules/1.1) 规范，
- `es` 表示构建出来的文件遵循 [ES Module](http://exploringjs.com/es6/ch_modules.html) 规范。
- `umd` 表示构建出来的文件遵循 [UMD](https://github.com/umdjs/umd) 规范。

以 `web-runtime-cjs` 配置为例，它的 `entry` 是 `resolve('web/entry-runtime.js')`；继续往下看`resolve` 函数的作用， 将传如的路径通过/分割成数组，然后去第一个元素设置为base；比如：

```js
const aliases = require("./alias");
const resolve = (p) => {
  // 将传入的路径通过/分割成数组，然后去第一个元素设置为base；base 并不是实际的路径，它的真实路径借助了aliases的配置
  const base = p.split("/")[0];
  if (aliases[base]) {
    return path.resolve(aliases[base], p.slice(base.length + 1));
  } else {
    return path.resolve(__dirname, "../", p);
  }
};
resolve('web/entry-runtime.js'); // 此时 base 则为 web
```

接着来看看`scripts/alias.js`文件的逻辑：

```js
const path = require('path')

const resolve = p => path.resolve(__dirname, '../', p)

module.exports = {
  vue: resolve('src/platforms/web/entry-runtime-with-compiler'),
  compiler: resolve('src/compiler'),
  core: resolve('src/core'),
  shared: resolve('src/shared'),
  web: resolve('src/platforms/web'),
  weex: resolve('src/platforms/weex'),
  server: resolve('src/server'),
  entries: resolve('src/entries'),
  sfc: resolve('src/sfc')
}
```

根据上面的例子，web对应的真实路径是`src/platforms/web`，然后`resolve()`函数通过`path.resolve(aliases[base], p.slice(base.length + 1))`找到最终路径，，它就是 Vue.js 源码 web 目录下的 `entry-runtime.js`；因此，`web-runtime-cjs` 配置对应的入口文件就找到了。它经过 Rollup 的构建打包后，最终会在 dist 目录下生成 `vue.runtime.common.js`
